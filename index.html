async function runPistonAPI() { const userCode = editor.getValue(); const p = probs[activeIdx]; const bar = document.getElementById('pBar'); const res = document.getElementById('vResult'); const btn = document.getElementById('sBtn'); document.querySelector('.progress').style.display = 'block'; bar.style.width = '0%'; res.innerText = "퍼블릭 서버로 코드 전송 중..."; res.className = "verdict text-warning"; btn.disabled = true; for (let i = 0; i < p.cases.length; i++) { const c = p.cases[i]; try { res.innerText = `케이스 #${i+1} : 원격 채점 중...`; // Piston API 호출 const submitRes = await fetch("https://emkc.org/api/v2/piston/execute", { method: "POST", headers: { "Content-Type": "application/json" // 만약 API 키를 발급받으셨다면 여기에 추가: "Authorization": "YOUR_KEY" }, body: JSON.stringify({ language: "python", version: "3.10.0", // '*' 보다는 구체적인 버전(3.10.0 등)을 쓰는 것이 안전합니다. files: [{ content: userCode }], stdin: c.in }) }); if (!submitRes.ok) { const errorData = await submitRes.json(); // 401 에러 시 서버의 메시지를 출력해 원인을 더 정확히 파악합니다. throw new Error(errorData.message || `서버 거부 (${submitRes.status})`); } const data = await submitRes.json(); // Piston API 응답 구조 확인 const actualOut = data.run.output ? data.run.output.trim() : ""; const actualErr = data.run.stderr ? data.run.stderr.trim() : ""; if (actualErr) { res.innerText = `런타임 에러 - 케이스 #${i+1}`; res.className = "verdict text-danger"; console.error("Runtime Error:", actualErr); btn.disabled = false; return; } // 결과 비교 (공백 제거 포함) if (actualOut === c.out.trim()) { let pct = Math.floor(((i + 1) / p.cases.length) * 100); bar.style.width = pct + '%'; } else { res.innerText = `틀렸습니다 - 케이스 #${i+1}`; res.className = "verdict text-danger"; console.log(`기대값: [${c.out}], 실제출력: [${actualOut}]`); btn.disabled = false; return; } // 서버 부하 방지를 위한 약간의 딜레이 await new Promise(r => setTimeout(r, 300)); } catch (e) { res.innerText = "에러: " + e.message; res.className = "verdict text-danger"; btn.disabled = false; return; } } res.innerText = "맞았습니다!! 🎉"; res.className = "verdict text-success"; btn.disabled = false; }
