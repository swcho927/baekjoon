<script>
    let pyodide;

    // 1. 파이썬 환경 준비
    async function setup() {
        pyodide = await loadPyodide();
        document.getElementById('status').innerText = "✅ 준비 완료!";
        document.getElementById('runBtn').disabled = false;
    }
    setup();

    // 2. 채점 로직
    async function runJudge() {
        const code = document.getElementById('codeTextarea').value;
        const resultBox = document.getElementById('resultBox');
        const resultText = document.getElementById('resultText');
        
        resultBox.style.display = "block";
        resultBox.className = "result-box bg-warning text-dark";
        resultText.innerText = "채점 중...";

        try {
            // 테스트케이스 설정 (입력값: 1 2)
            pyodide.runPython(`
                import io, sys
                sys.stdin = io.StringIO("1 2")
                sys.stdout = io.StringIO()
            `);

            // 코드 실행
            await pyodide.runPythonAsync(code);

            // 결과값 가져오기
            const output = pyodide.runPython("sys.stdout.getvalue()").trim();
            
            // 정답 비교 (정답: 3)
            if (output === "3") {
                resultBox.className = "result-box bg-success text-white";
                resultText.innerText = "맞았습니다!!";
            } else {
                resultBox.className = "result-box bg-danger text-white";
                resultText.innerText = `틀렸습니다 (내 출력: ${output})`;
            }
        } catch (err) {
            resultBox.className = "result-box bg-dark text-white";
            resultText.innerText = "에러 발생: " + err.message;
        }
    }

    document.getElementById('runBtn').onclick = runJudge;
</script>
